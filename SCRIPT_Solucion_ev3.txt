-----------SCRIPT DE CONSULTAS para responder Preguntas de la EVALUACION----------

--Pregunta N°1. Número de evaluaciones creadas por programa(curso).

SELECT PROGRAMA.NOMBRE AS PROGRAMA, COUNT(TEST_ID) AS NUM_EVALUACIONES_CREADAS FROM TEST
INNER JOIN UNIDAD USING (UNIDAD_ID)
INNER JOIN PROGRAMA ON (UNIDAD.PROGRAMA_ID=PROGRAMA.PROGRAMA_ID)
GROUP BY PROGRAMA.NOMBRE;

--Pregunta N°2. Módulos por programa sin evaluaciones.

SELECT UNIDAD.UNIDAD_ID, UNIDAD.NOMBRE AS MODULOS_SIN_EVALUACIONES, PROGRAMA.NOMBRE FROM UNIDAD
INNER JOIN PROGRAMA ON UNIDAD.PROGRAMA_ID=PROGRAMA.PROGRAMA_ID
WHERE UNIDAD_ID NOT IN (SELECT TEST.UNIDAD_ID FROM TEST)
ORDER BY UNIDAD.UNIDAD_ID;

--Pregunta N°3. Evaluaciones creadas con deficiencia.

SELECT UNIDAD.NOMBRE AS programa, test.test_id as test_id_deficiente, test.nombre as nombre_unidad_deficiente FROM UNIDAD
INNER JOIN TEST ON UNIDAD.UNIDAD_ID=TEST.UNIDAD_ID
INNER JOIN PREGUNTA ON TEST.TEST_ID=PREGUNTA.TEST_ID
WHERE PREGUNTA.ENUNCIADO=' ' OR pregunta.CANT_ALTERNATIVAS<=2 OR PREGUNTA.PUNTAJE_ASOC>=100 OR PREGUNTA.PUNTAJE_ASOC=0;

--Pregunta N°4. Cantidad de alumnos por curso.

SELECT PROGRAMA.NOMBRE AS PROGRAMA, COUNT(ESTUDIANTE_ID) AS CANT_ALUMNOS FROM ESTUDIANTE
INNER JOIN PROGRAMA USING (PROGRAMA_ID)
GROUP BY PROGRAMA.NOMBRE;

--Pregunta N°5. Puntaje NO Normalizado de cada evaluación rendida.

SELECT RESPUESTA_TEST.ESTUDIANTE_ID, SUM(ALTERNATIVA.PONDERACION)*10
-(SELECT COUNT(ALTERNATIVA.ALTERNATIVA_ID)/4 FROM ALTERNATIVA
INNER JOIN RESPUESTA_TEST ON ALTERNATIVA.PREGUNTA_ID=RESPUESTA_TEST.PREGUNTA_ID
WHERE ALTERNATIVA.VALOR_LOGICO='F' AND ALTERNATIVA.ALTERNATIVA_ID=RESPUESTA_TEST.ALTERNATIVA_ID)AS PUNTAJE_NO_NORMALIZADO
FROM RESPUESTA_TEST
INNER JOIN ALTERNATIVA 
ON RESPUESTA_TEST.ALTERNATIVA_ID=ALTERNATIVA.ALTERNATIVA_ID
GROUP BY RESPUESTA_TEST.ESTUDIANTE_ID
ORDER BY RESPUESTA_TEST.ESTUDIANTE_ID;

--Pregunta N°6. Nota Normalizada (escala de 1,0 a 7,0).

SELECT RESPUESTA_TEST.ESTUDIANTE_ID, ROUND(((SUM(ALTERNATIVA.PONDERACION)*10
-(SELECT COUNT(ALTERNATIVA.ALTERNATIVA_ID)/4 FROM ALTERNATIVA
INNER JOIN RESPUESTA_TEST ON ALTERNATIVA.PREGUNTA_ID=RESPUESTA_TEST.PREGUNTA_ID
WHERE ALTERNATIVA.VALOR_LOGICO='F' AND ALTERNATIVA.ALTERNATIVA_ID=RESPUESTA_TEST.ALTERNATIVA_ID)-60)*3/40)+4, 1) AS NOTA_NORMALIZADA
FROM RESPUESTA_TEST
INNER JOIN ALTERNATIVA 
ON RESPUESTA_TEST.ALTERNATIVA_ID=ALTERNATIVA.ALTERNATIVA_ID
GROUP BY RESPUESTA_TEST.ESTUDIANTE_ID;

--Respuesta Pregunta N°7

SELECT RESPUESTA_TEST.ESTUDIANTE_ID, ROUND(((SUM(ALTERNATIVA.PONDERACION)*10
-(SELECT COUNT(ALTERNATIVA.ALTERNATIVA_ID)/4 FROM ALTERNATIVA
INNER JOIN RESPUESTA_TEST ON ALTERNATIVA.PREGUNTA_ID=RESPUESTA_TEST.PREGUNTA_ID
WHERE ALTERNATIVA.VALOR_LOGICO='F' AND ALTERNATIVA.ALTERNATIVA_ID=RESPUESTA_TEST.ALTERNATIVA_ID)-60)*3/40)+4, 1) AS NOTA_NORMALIZADA
FROM RESPUESTA_TEST
INNER JOIN ALTERNATIVA 
ON RESPUESTA_TEST.ALTERNATIVA_ID=ALTERNATIVA.ALTERNATIVA_ID
INNER JOIN ESTUDIANTE ON RESPUESTA_TEST.ESTUDIANTE_ID=ESTUDIANTE.ESTUDIANTE_ID
GROUP BY RESPUESTA_TEST.ESTUDIANTE_ID
ORDER BY RESPUESTA_TEST.ESTUDIANTE_ID;

--PREGUNTA 8 EN PROCESO

SELECT AVG((ROUND(((SUM(ALTERNATIVA.PONDERACION)*10
-(SELECT COUNT(ALTERNATIVA.ALTERNATIVA_ID)/4 FROM ALTERNATIVA
INNER JOIN RESPUESTA_TEST ON ALTERNATIVA.PREGUNTA_ID=RESPUESTA_TEST.PREGUNTA_ID
WHERE ALTERNATIVA.VALOR_LOGICO='F' AND ALTERNATIVA.ALTERNATIVA_ID=RESPUESTA_TEST.ALTERNATIVA_ID)-60)*3/40)+4, 1)))
AS PROMEDIO_NOTAS
FROM RESPUESTA_TEST
INNER JOIN ALTERNATIVA 
ON RESPUESTA_TEST.ALTERNATIVA_ID=ALTERNATIVA.ALTERNATIVA_ID
inner join pregunta on alternativa.pregunta_id=pregunta.pregunta_id
inner join test on pregunta.test_id=test.test_id
inner join unidad on test.unidad_id=unidad.unidad_id
inner join programa on unidad.programa_id=programa.programa_id
group BY respuesta_test.TEST_id;
